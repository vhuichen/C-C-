poetry

poetry

提供了依赖管理和打包功能

安装

# 方式一（新手不建议，不知道安装目录在哪里）
# 默认安装位置：~/.local/bin/poetry
curl -sSL https://install.python-poetry.org | python3 -
export PATH="~/.local/bin:$PATH"

# 方式二：可以在自己指定的 Python 环境中安装
# 直接 pip 安装
pip install poetry

#
poetry --version

poetry 默认会将虚拟环境统一放在指定目录，具体信息可以通过 poetry config --list 查看

cache-dir = "~/Library/Caches/pypoetry"
experimental.system-git-client = false
installer.max-workers = null
installer.modern-installation = true
installer.no-binary = null
installer.parallel = true
keyring.enabled = true
solver.lazy-wheel = true
virtualenvs.create = true
virtualenvs.in-project = null
virtualenvs.options.always-copy = false
virtualenvs.options.no-pip = false
virtualenvs.options.no-setuptools = false
virtualenvs.options.system-site-packages = false
virtualenvs.path = "{cache-dir}/virtualenvs"  # ~/Library/Caches/pypoetry/virtualenvs
virtualenvs.prefer-active-python = false
virtualenvs.prompt = "{project_name}-py{python_version}"
warnings.export = true

如果要将虚拟项目存放在当前项目目录呢？可以用一下命令

poetry config virtualenvs.in-project true

命令

# 当前目录使用的 Python 版本
# 命令会根据指定的版本在项目中创建一个新的虚拟环境（如果尚不存在的话），或者切换到一个已经存在的虚拟环境。这个虚拟环境将使用你指定的 Python 版本。
poetry env use 3.10.14
功能 示例
添加依赖 poetry add xxx
移除依赖 poetry remove xxx
导出依赖 poetry export -f requirements.txt
列出项目依赖详细信息 poetry show xxx
指定 Python 解释器 poetry env use python3.10 / poetry env use 3.10.14
启动虚拟环境 poetry shell
退出虚拟环境变量 exit （pycharm 退出好像有问题）
初始化 poetry init
安装依赖 poetry install
创建新的 Poetry 项目 poetry new xxxx
打包项目 poetry build
（同步）更新 lock 文件 poetry lock --no-update
发布项目到PyPI poetry publish --username --password
修改源 poetry source add tsinghua https://pypi.tuna.tsinghua.edu.cn/simple
列出相关配置信息 poetry config --list
更新版本 poetry self update

项目文件

  • pyproject.toml；用于配置和管理一个Python项目
  • poetry.lock；详细记录了所有安装的模块与版本

pyproject.toml

[tool.poetry]
name = "poetry-demo"
version = "0.1.0"
description = ""
authors = ["wangss <913292836@qq.com>"]
readme = "README.md"

[tool.poetry.dependencies]
python = "3.10.14" # 固定版本
numpy = ">=1.21.1" # 必须大于等于 1.21.1 版本
Cython = "^0.29" # 必须大于等于 0.29 版本，并必须要与 0.29 兼容
my_pack = { path = "./xxx/my_pack-0.1.0-py3-none-any.whl" } # 添加一个本地包依赖

[build-system]
requires = ["poetry-core"]
build-backend = "poetry.core.masonry.api"
