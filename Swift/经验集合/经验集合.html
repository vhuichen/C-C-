UIPickerView

UIPickerView {
      // 设置中间选中View的背景色
    // 已适配系统：14.2
    func setSelectViewBgColor(_ color: UIColor) {
        guard let contentView = self.subviews.first else {
            return
        }
          //只需要设置第一个就行了
        guard let firstColumnView = contentView.subviews.filter({
            String(describing: type(of: $0)) == "UIPickerColumnView"
        }).first
        else {
            return
        }
        guard let middleView = firstColumnView.value(forKey: "middleContainerView") as? UIView else {
            return
        }
        middleView.backgroundColor = color
    }
}
// pickerView 自定义样式
func pickerView(
    _: UIPickerView,
    viewForRow row: Int,
    forComponent component: Int,
    reusing view: UIView?
) -> UIView {
    var label: UILabel!
    if let view = view as? UILabel {
        label = view
    } else {
        label = UILabel()
        label.textColor = .c3Color
        label.font = .t1Font
        label.textAlignment = .center
    }
    label.text = "123"

    DispatchQueue.main.async {
        if let label = self.pickerView.view(forRow: row, forComponent: component) as? UILabel {
              // 自定义选中View的字体颜色
            label.textColor = UIColor.red
            if component == 0 {
                self.pickerView.setSelectViewBgColor(UIColor.green)
            }
        }
    }
    return label
}

UINavigationBar

// 1. 子 View 的第一个View一般是 _UIBarBackground
// 2. _UIBarBackground 的高度不一定等于导航栏+状态栏的高度（iOS 17.4）
// 3. viewDidLoad内部，UINavigationBar可能还没布局

/// 获取 _UIBarBackground
/// 已验证适配系统：14.2、15.0、16.4、17.2
func navigationBarBgView() -> UIView? {
    guard let navigationBar = self.navigationController?.navigationBar else {
        return nil
    }
    var bgView: UIView?
    for view in navigationBar.subviews {
        let className = String(describing: type(of: view))
        if className.hasSuffix("BarBackground") {
            bgView = view
            break
        }
    }
    return bgView
}

判断主队列

// 判断是否是主队列（RxSwift 代码）
// NSThread.isMainThread 是用来判断是否是主线程，但主线程执行的任务不一定是主队列的任务；
// 对于一些 UIView（例如：MKMapView），需要确保在主队列执行，而不仅仅是主线程
// 参考：
// https://www.51cto.com/article/534499.html
// https://www.jianshu.com/p/7f68a3d5b07d
// https://github.com/ReactiveCocoa/ReactiveCocoa/pull/2912
// 疑问：如果别的队列也使用这个 key 会怎样
public extension DispatchQueue {
    private static var token: DispatchSpecificKey<Void> = {
        let key = DispatchSpecificKey<Void>()
        DispatchQueue.main.setSpecific(key: key, value: ())
        return key
    }()
    static var isMain: Bool {
        DispatchQueue.getSpecific(key: token) != nil
    }
}
