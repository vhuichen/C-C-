属性包装器

属性包装器

@ObservedObject & @Published

ObservedObject 意为监听一个对象，Published 意为需要监听的属性

class Person: ObservableObject {
    @Published var count = 0
}
struct TempExample: View {
    @ObservedObject var p = Person()
    var body: some View {
        VStack {
            Text("\(p.count)")
            .padding()
            Button("点我+1") {
                p.count += 1
            }
        }
    }
}

@StateObject

StateObject 的生命周期与当前 View 生命周期一致，即当 View 被销毁后，StateObject 的数据才会销毁，当 View 被刷新时，StateObject 的数据会保持。而 ObservedObject 不被 View 持有，生命周期不一定与 View 一致，页面刷新时数据可能会销毁。

class Person: ObservableObject {
    @Published var count = 0
    deinit{
        print("ObservableObject 销毁了")
    }
}
struct TempExample: View {
    @State var count = 0
    var body: some View {
        VStack{
            Text("CounterView:\(count)")
            Button("刷新") {
                count += 1
            }
            MapView()
        }
    }
}
struct MapView: View {
    // StateObject页面刷新时，对象不会销毁，不会调用 deinit
    @StateObject var p = Person()
    // ObservedObject 页面刷新时，对象会销毁，会调用 deinit
    // @ObservedObject var p = Person()
    var body: some View {
        VStack{
            Text("\(p.count)")
            Button("点击+1") {
                p.count += 1
            }

        }
    }
}

@EnvironmentObject

用于跨多个视图层级共享状态或数据。

class UserSettings: ObservableObject {
    @Published var count = 0
}
struct TempExample: View {
    @EnvironmentObject var userSettings: UserSettings
    var body: some View {
        VStack {
            Text("Current: \(userSettings.count)").padding(10)
            Button("点我+1") {
                userSettings.count += 1
            }
        }
    }
}
struct TempExample_Previews: PreviewProvider {
    static var userSettings = UserSettings()
    static var previews: some View {
        TempExample()
            .environmentObject(userSettings)
    }
}

@Environment

用于访问环境值，用于只读访问环境值，会绑定数据更新

struct ContentView: View {
  // 使用 @Environment 访问当前的颜色方案
  @Environment(\.colorScheme) var colorScheme: ColorScheme
  var body: some View {
    Text("Current Color Scheme: \(colorScheme == .dark ? "Dark" : "Light")")
  }
}

@Binding

将父视图的状态传递给子视图，使子视图能够读取和修改该状态

struct TempExample: View {
    @State var count = 0
    var body: some View {
        SubView(subCount: $count)
        Text("\(count)")
            .padding()
            .background(.blue)
        Button("点我加1") {
          self.count += 1
        }
    }
}
struct SubView: View {
  @Binding var subCount: Int
    var body: some View {
        Button("点我加10") {
          self.subCount += 10
        }
        Text("\(subCount)")
            .padding()
            .background(.red)
    }
}

@State

修饰的属性存储在特殊的内存区域中，当值发生变化时， SwiftUI 可以检测到该变化并相应地更新 对应的 View。

必须是在 View 内部的属性使用（不然怎么知道该更新哪个 View）
